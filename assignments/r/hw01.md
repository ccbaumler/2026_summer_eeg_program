    ```R

    tail()    # preview the bottom of the data
    nrow()    # to get row count of the data
    ncol()    # to get the column count of the data
    dim()     # to get the row and column count of the data
    length()  # to get length of the data
    names()   # to get the names of the object
    colnames()# to get the column names of the object
    rownames()# to get the row names of the object
    class()   # to get the class of the object
    ```
